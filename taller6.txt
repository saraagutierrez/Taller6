hereda_propiedad(Clase, Propiedad, Valor) :-
    propiedad(Clase, Propiedad, Valor).
    
hereda_propiedad(clase, propiedad, Valor) :-
    inst(clase, superclase1),
    es_un(clase, superclase2),
    hereda_propiedad(superclase2, propiedad, Valor);
	hereda_propiedad(superclase1, propiedad, Valor).
